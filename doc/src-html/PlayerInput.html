010import java.util.Scanner;
011import java.util.InputMismatchException;
012
013public class PlayerInput {
014    private static Scanner input = new Scanner(System.in);
015
016    /**
017     * Asks the player a place to shoot on the playing field.
018     * <p>
019     * Asks the player for a letter and a number to know which spot on the
020     * playing field will be shot.
021     * 
022     * @param x
023     *            the board to be shot.
024     * @param shipArray
025     *            and array of Ships located on the board.
026     */
027    public static void shoot(Board x, Ship[] shipArray) {
028        String[] letterArray = { "a", "b", "c", "d", "e", "f", "g", "h", "i",
029                "j" };
030        boolean again = true;
031        String letter = "";
032        int number = 0;
033
034        System.out
035                .println("Choose a spot to bomb by entering the corresponding number and letter");
036
037        do {
038            System.out.println("Choose a letter");
039            letter = input.nextLine();
040
041            for (int i = 0; i < letterArray.length; i++) {
042                if (letter.equals(letterArray[i])) {
043                    again = false;
044                }
045            }
046            if (again) {
047                System.out.println("This letter is not valid");
048            }
049        } while (again);
050        int letterInt = letterToNumber(letter);
051        again = true;
052
053        do {
054            try {
055                System.out.println("Choose a number");
056                number = input.nextInt();
057                input.nextLine();
058                again = false;
059                if (number > 10 || number < 1) {
060                    System.out.println("This number is not valid");
061                    again = true;
062                }
063            } catch (InputMismatchException e) {
064                System.out.println("This number is not valid");
065                input.nextLine();
066            }
067        } while (again);
068
069        x.hitOrMiss(number - 1, letterInt, shipArray);
070        x.shipDeath(shipArray);
071    }
072
073    /**
074     * Changes a letter to a number.
075     * <p>
076     * Takes a valid input of a letter from A to J and changes it to a number
077     * representing it on the playing field array.
078     * 
079     * @param x
080     *            the letter to be changed.
081     * @return an integer from 0 to 9.
082     */
083    public static int letterToNumber(String x) {
084        switch (x) {
085        case "a":
086            return 0;
087        case "b":
088            return 1;
089        case "c":
090            return 2;
091        case "d":
092            return 3;
093        case "e":
094            return 4;
095        case "f":
096            return 5;
097        case "g":
098            return 6;
099        case "h":
100            return 7;
101        case "i":
102            return 8;
103        case "j":
104            return 9;
105        }
106        return 0;
107    }

109    /**
110     * Asks the player for a spot to place a ship.
111     * <p>
112     * Asks for a letter and a number representing the spot on the playing field
113     * from where the ship will be started from and follows up with a question
114     * for the direction to which the ship will be built.
115     * 
116     * @param x
117     *            the board to which the ship will be placed.
118     * @param shipArray
119     *            the array of Ship objects to be placed.
120     */
121    public static void chooseShips(Board x, Ship[] shipArray) {
122
123        String[] array = { "", "first ", "second ", "", "" };
124        String[] letterArray = { "a", "b", "c", "d", "e", "f", "g", "h", "i",
125                "j" };
126        boolean again = true;
127        String letter = "";
128        int number = 0;
129        String direction = "";
130        int letterInt = 0;
131
132        boolean again2 = false;
133        boolean again3 = false;
134
135        do {
136            again2 = false;
137
138            System.out
139                    .println("Do you want to choose where your ships are yourself? Y/N");
140            String notRandom = input.nextLine();
141
142            if (notRandom.equalsIgnoreCase("Y")) {
143
144                x.printBoard(true);
145                System.out.println();
146
147                for (int k = 0; k < shipArray.length; k++) {
148                    do {
149
150                        System.out.println("Choose where you want your "
151                                + array[k] + "size " + shipArray[k].getSize()
152                                + " ship to start from.");
153
154                        do {
155                            again = true;
156                            
157                            System.out.println("Choose a letter");
158                            letter = input.nextLine();
159
160                            for (int i = 0; i < letterArray.length; i++) {
161                                if (letter.equals(letterArray[i])) {
162                                    again = false;
163                                }
164                            }
165                            if (again) {
166                                System.out.println("This letter is not valid");
167                            }
168                        } while (again);
169                        letterInt = letterToNumber(letter);
170                        again = true;
171
172                        do {
173                            try {
174                                System.out.println("Choose a number");
175                                number = input.nextInt() - 1;
176                                input.nextLine();
177                                again = false;
178                                if (number > 9 || number < 0) {
179                                    System.out
180                                            .println("This number is not valid");
181                                    again = true;
182                                }
183                            } catch (InputMismatchException e) {
184                                System.out.println("This number is not valid");
185                                input.nextLine();
186                            }
187                        } while (again);
188
189                        do {
190                            System.out
191                                    .println("Choose which direction you want the ship to extend.");
192                            direction = input.nextLine();
193
194                            if (direction.equalsIgnoreCase("w")
195                                    || direction.equalsIgnoreCase("a")
196                                    || direction.equalsIgnoreCase("s")
197                                    || direction.equalsIgnoreCase("d")) {
198                                again3 = false;
199                            } else {
200                                System.out.println("This input is not valid.");
201                                again3 = true;
202                            }
203                        } while (again3);
204                    } while (Exception.doesShipFit(x, shipArray[k].getSize(),
205                            number, letterInt, direction) == false);
206
207                    shipArray[k].chosenShipLocation(x, number, letterInt,
208                            direction, shipArray[k].getSize());
209                }
210
211            } else if (notRandom.equalsIgnoreCase("N")) {
212
213                shipArray[0].randomlyPlaceShip(x);
214                shipArray[1].randomlyPlaceShip(x);
215                shipArray[2].randomlyPlaceShip(x);
216                shipArray[3].randomlyPlaceShip(x);
217                shipArray[4].randomlyPlaceShip(x);
218
219            } else {
220                System.out.println("This on not a valid input.");
221                again2 = true;
222            }
223        } while (again2);
224
225        System.out.println("      GAME START!\n");
226    }
