010import java.lang.ArrayIndexOutOfBoundsException;
011
012public class Exception {
013    /**
014     * Checks it there is an X around the specified spot in an array.
015     * 
016     * @param x
017     *            the board to where a ship will be added.
018     * @param posY
019     *            the Y axis position on the board.
020     * @param posX
021     *            the X axis position on the board.
022     * @return true if there is an X around the spot.
023     */
024    public static boolean shipPlacement(Board x, int posY, int posX) {
025
026        try {
027            if (x.getPlayingField()[posY - 1][posX - 1] == " X"
028                    || x.getPlayingField()[posY - 1][posX] == " X"
029                    || x.getPlayingField()[posY - 1][posX + 1] == " X"
030                    || x.getPlayingField()[posY][posX - 1] == " X"
031                    || x.getPlayingField()[posY][posX + 1] == " X"
032                    || x.getPlayingField()[posY + 1][posX - 1] == " X"
033                    || x.getPlayingField()[posY + 1][posX] == " X"
034                    || x.getPlayingField()[posY + 1][posX + 1] == " X") {
035                return true;
036            }
037        } catch (ArrayIndexOutOfBoundsException e) {
038
039            try {
040                if (posY == 0) {
041                    if (posX == 0) {
042                        if (x.getPlayingField()[posY][posX + 1] == " X"
043                                || x.getPlayingField()[posY + 1][posX] == " X"
044                                || x.getPlayingField()[posY + 1][posX + 1] == " X") {
045                            return true;
046                        }
047                    } else if (posX == x.getPlayingField().length - 1) {
048                        if (x.getPlayingField()[posY][posX - 1] == " X"
049                                || x.getPlayingField()[posY + 1][posX] == " X"
050                                || x.getPlayingField()[posY + 1][posX - 1] == " X") {
051                            return true;
052                        }
053                    } else {
054                        if (x.getPlayingField()[posY][posX + 1] == " X"
055                                || x.getPlayingField()[posY + 1][posX] == " X"
056                                || x.getPlayingField()[posY][posX - 1] == " X"
057                                || x.getPlayingField()[posY + 1][posX + 1] == " X"
058                                || x.getPlayingField()[posY + 1][posX - 1] == " X") {
059                            return true;
060                        }
061
062                    }
063                } else if (posY == x.getPlayingField().length - 1) {
064                    if (posX == 0) {
065                        if (x.getPlayingField()[posY][posX + 1] == " X"
066                                || x.getPlayingField()[posY - 1][posX] == " X"
067                                || x.getPlayingField()[posY - 1][posX + 1] == " X") {
068                            return true;
069                        }
070                    } else if (posX == x.getPlayingField().length - 1) {
071                        if (x.getPlayingField()[posY][posX - 1] == " X"
072                                || x.getPlayingField()[posY - 1][posX] == " X"
073                                || x.getPlayingField()[posY - 1][posX - 1] == " X") {
074                            return true;
075                        }
076                    } else {
077                        if (x.getPlayingField()[posY][posX + 1] == " X"
078                                || x.getPlayingField()[posY - 1][posX] == " X"
079                                || x.getPlayingField()[posY][posX - 1] == " X"
080                                || x.getPlayingField()[posY - 1][posX + 1] == " X"
081                                || x.getPlayingField()[posY - 1][posX - 1] == " X") {
082                            return true;
083                        }
084                    }
085                } else if (posX == 0) {
086                    if (x.getPlayingField()[posY][posX + 1] == " X"
087                            || x.getPlayingField()[posY - 1][posX] == " X"
088                            || x.getPlayingField()[posY + 1][posX] == " X"
089                            || x.getPlayingField()[posY + 1][posX + 1] == " X"
090                            || x.getPlayingField()[posY - 1][posX + 1] == " X") {
091                        return true;
092                    }
093                } else if (posX == x.getPlayingField().length - 1) {
094                    if (x.getPlayingField()[posY][posX - 1] == " X"
095                            || x.getPlayingField()[posY - 1][posX] == " X"
096                            || x.getPlayingField()[posY + 1][posX] == " X"
097                            || x.getPlayingField()[posY + 1][posX - 1] == " X"
098                            || x.getPlayingField()[posY - 1][posX - 1] == " X") {
099                        return true;
100                    }
101                } else {
102                    return false;
103                }
104            } catch (ArrayIndexOutOfBoundsException h) {
105            }
106            ;
107        }
108        ;
109
110        return false;
111    }
112
113    /**
114     * Removes a ship which has been placed to an invalid location.
115     * <p>
116     * Chances the unwanted X:s marking the ship back to empty spots on the
117     * playing field.
118     * 
119     * @param x
120     *            the board from which the ship will be removed.
121     * @param direction
122     *            the direction to which the unwanted ship was placed.
123     * @param length
124     *            the length of the unwanted ship.
125     */
126    public static void removeFailShip(Board x, int direction, int length,
127            int posY, int posX) {
128        int removeLength = length - 1;
129
130        for (int i = 0; i < length - 1; i++) {
131            switch (direction) {
132            case 0: // 'W'
133
134                x.getPlayingField()[posY - removeLength][posX] = "[]";
135
136                break;
137            case 1: // 'A'
138
139                x.getPlayingField()[posY][posX - removeLength] = "[]";
140
141                break;
142            case 2: // 'S'
143
144                x.getPlayingField()[posY + removeLength][posX] = "[]";
145
146                break;
147            case 3: // 'D'
148
149                x.getPlayingField()[posY][posX + removeLength] = "[]";
150
151                break;
152            }
153            removeLength--;
154        }
155    }

157    /**
158     * Checks if a ship fits in a spot chosen by the player.
159     * 
160     * @param x
161     *            the board to which the ship will be placed.
162     * @param shipLength
163     *            the lenght of the ship.
164     * @param posY
165     *            the Y axis position on the board.
166     * @param posX
167     *            the X axis position on the board.
168     * @param direction
169     *            the direction of the ship.
170     */
171    public static boolean doesShipFit(Board x, int shipLength, int posY,
172            int posX, String direction) {
173
174        try {
175            if (x.getPlayingField()[posY][posX] == "[]") {
176
177                if (shipPlacement(x, posY, posX) == false) {
178
179                    for (int i = 1; i < shipLength; i++) {
180                        switch (direction) {
181                        case "w":
182                            if (x.getPlayingField()[posY - i][posX] == "[]") {
183                                if (shipPlacement(x, posY - i, posX)) {
184                                    System.out
185                                            .println("The spot where you wanted to place your ship was not big enough.");
186                                    return false;
187                                }
188                            }
189                            break;
190                        case "a":
191                            if (x.getPlayingField()[posY][posX - i] == "[]") {
192                                if (shipPlacement(x, posY, posX - i)) {
193                                    System.out
194                                            .println("The spot where you wanted to place your ship was not big enough.");
195                                    return false;
196                                }
197                            }
198                            break;
199                        case "s":
200                            if (x.getPlayingField()[posY + 1][posX] == "[]") {
201                                if (shipPlacement(x, posY + i, posX)) {
202                                    System.out
203                                            .println("The spot where you wanted to place your ship was not big enough.");
204                                    return false;
205                                }
206                            }
207                            break;
208                        case "d":
209                            if (x.getPlayingField()[posY][posX + 1] == "[]") {
210                                if (shipPlacement(x, posY, posX + i)) {
211                                    System.out
212                                            .println("The spot where you wanted to place your ship was not big enough.");
213                                    return false;
214                                }
215                            }
216                            break;
217                        }
218
219                    }
220                    return true;
221                }
222            }
223        } catch (ArrayIndexOutOfBoundsException e) {
224            System.out
225                    .println("The spot where you wanted to place your ship was not big enough.");
226            return false;
227        }
228
229        System.out
230                .println("The spot where you wanted to place your ship was not big enough.");
231        return false;
232    }
