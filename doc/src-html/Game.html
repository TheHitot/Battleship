001/**

003 *

008 */
009public class Game {
010
011    private static Board playerField;
012    private static Board enemyField;
013    private static AI computer;
014    private static Ship[] enemyShipArray;
015    private static Ship[] playerShipArray;
016
017    /**
018     * Creates all the necessary objects for the game.
019     * <p>
020     * Creates the starting fields and all the Objects needed to run the game.
021     * 
022     */
023    public static void createGame() {
024        // TODO Auto-generated method stub
025
026        playerField = new Board(10);
027        playerField.createBoard();
028        enemyField = new Board(10);
029        enemyField.createBoard();
030        Ship enemyShip2 = new Ship(2);
031        Ship enemyShip3_1 = new Ship(3);
032        Ship enemyShip3_2 = new Ship(3);
033        Ship enemyShip4 = new Ship(4);
034        Ship enemyShip5 = new Ship(5);
035
036        Ship playerShip2 = new Ship(2);
037        Ship playerShip3_1 = new Ship(3);
038        Ship playerShip3_2 = new Ship(3);
039        Ship playerShip4 = new Ship(4);
040        Ship playerShip5 = new Ship(5);
041
042        /*
043         * //In case of an emergency! Ship[] enemyShipArray = { enemyShip2,
044         * enemyShip3_1, enemyShip3_2, enemyShip4, enemyShip5 }; Ship[]
045         * playerShipArray = { playerShip2, playerShip3_1, playerShip3_2,
046         * playerShip4, playerShip5 };
047         */
048        enemyShipArray = new Ship[5];
049        playerShipArray = new Ship[5];
050
051        enemyShipArray[0] = enemyShip2;
052        enemyShipArray[1] = enemyShip3_1;
053        enemyShipArray[2] = enemyShip3_2;
054        enemyShipArray[3] = enemyShip4;
055        enemyShipArray[4] = enemyShip5;
056
057        playerShipArray[0] = playerShip2;
058        playerShipArray[1] = playerShip3_1;
059        playerShipArray[2] = playerShip3_2;
060        playerShipArray[3] = playerShip4;
061        playerShipArray[4] = playerShip5;
062
063        enemyShip5.randomlyPlaceShip(enemyField);
064        enemyShip4.randomlyPlaceShip(enemyField);
065        enemyShip3_2.randomlyPlaceShip(enemyField);
066        enemyShip3_1.randomlyPlaceShip(enemyField);
067        enemyShip2.randomlyPlaceShip(enemyField);
068
069        PlayerInput.chooseShips(playerField, playerShipArray);
070
071        computer = new AI();
072
073        System.out.println("      Your field");
074        playerField.printBoard(true);
075        // System.out.println("\n   Computer's field");
076        // enemyField.printBoard(true);
077        System.out.println();
078        System.out.println("\n   Computer's field");
079        enemyField.printBoard(false);
080        System.out.println();
081        System.out.println();
082    }
083
084    /**
085     * Runs the game.
086     * <p>
087     * Runs player and AI decisions for the game and keeps the game running
088     * until on side has won.
089     * 
090     */
091
092    public static void startGame() {
093
094        do {
095            PlayerInput.shoot(enemyField, enemyShipArray);
096
097            if (enemyField.playerHasShot()) {
098                computer.shoot(playerField, playerShipArray);
099            }
100            System.out.println("      Your field");
101            playerField.printBoard(true);
102            // System.out.println("\n       Computer");
103            // enemyField.printBoard(true);
104            System.out.println();
105            System.out.println("\n   Computer's field");
106            enemyField.printBoard(false);
107            System.out.println();
108            System.out.println();
109
110        } while (enemyField.gameEnd(enemyShipArray) == false
111                && playerField.gameEnd(playerShipArray) == false);
112
113        System.out.println();
114
115        if (enemyField.gameEnd(enemyShipArray)) {
116            System.out.println("<><><><><><><><><><><><><><><><>");
117            System.out.println("<> CONGRATULATIONS, YOU WIN!  <>");
118            System.out.println("<><><><><><><><><><><><><><><><>");
119        } else {
120            System.out.println("~~~~~~~~~~~~~~~~~~~~~~~~");
121            System.out.println("|  Sorry, You lost...  |");
122            System.out.println("~~~~~~~~~~~~~~~~~~~~~~~~");
123        }
124    }
125
126}
